  \subsection{Floyd Warshall}
    \par \textbf{Usage:} Calculates all-pairs shortest path in a directed graph that might have negative edge weights.
    Input is an distance matrix $m$, where $m[i][j] = inf$ if $i$ and $j$ are not adjacent.
    As output, $m[i][j]$ is set to the shortest distance between $i$ and $j$, $inf$ if no path, or $-inf$ if the path goes thorugh a negative-weight cycle.
    \newline
    \par \textbf{Time Complexity:} $\mathcal{O}(n^3)$
    \inputminted{cpp}{library/graphs/floyd_warshall.cpp}

  \subsection{Euler's Totient Function}
    \par \textbf{Usage:} \emph{Euler's $\phi$} function is defined as $\phi(n):=\#$ of positive integers $\leq n$ that are coprime with $n$.
  $\phi(1)=1$, $p$ prime $\Rightarrow \phi(p^k)=(p-1)p^{k-1}$, $m,n$ coprime $\Rightarrow \phi(mn)=\phi(m)\phi(n)$.
  If $n=p_1^{k_1}p_2^{k_2} ... p_r^{k_r}$ then $\phi(n) = (p_1-1)p_1^{k_1-1}...(p_r-1)p_r^{k_r-1}$.
  $\phi(n)=n \cdot \prod_{p|n}(1-1/p)$.
      \textbf{Euler's thm}: $a,n$ coprime $\Rightarrow a^{\phi(n)} \equiv 1 \pmod{n}$.
     \textbf{Fermat's little thm}: $p$ prime $\Rightarrow a^{p-1} \equiv 1 \pmod{p}$ $\forall a$. 
    \par \textbf{Time Complexity:} $\mathcal{O}(\sqrt(n))$ / $\mathcal{O}(n\log\log(n))$
    \inputminted{cpp}{library/maths/euler_phi.hpp}

  \subsection{Manacher's algorithm}
    \par \textbf{Usage:}Computes palindrome radii on string $s$ interleaved with '\#'. Returns vector $d$ of size $2n+1$.
    \begin{itemize}
        \item $d[2i] = 2d_{even}[i] + 1$ (centered at '\#')
        \item $d[2i + 1] = 2d_{odd}[i]$ (centered at $s[i]$)
        \item Palindrome length = $d[i] - 1$.
        \item Substring $s[L \dots R]$ is a palindrome iff $d[L + R + 1] - 1 \ge R - L + 1$.
    \end{itemize}
    \\
    \par \textbf{Time Complexity:} $\mathcal{O}(n)$
    \inputminted{cpp}{library/strings/manacher.hpp}

\section{Game Theory \& Combinatorial Games}

\subsection{The Game Resolution Pipeline}
\par To solve an unfamiliar impartial game during a contest, follow this process:
\begin{enumerate}
    \item \textbf{Verify Impartiality:} Ensure both players have identical move options from any state, and it is a normal play game (last move wins).
    \item \textbf{Brute-Force SG Values:} Code a quick MEX script to calculate Grundy values ($G$) for small test inputs to detect structural patterns.
    \item \textbf{Analyze the Pattern:} Look for:
    \begin{itemize}
        \item \textbf{Periodicity:} $G(x) = G(x \pmod P)$ where $P$ is the cycle length.
        \item \textbf{Eliminable Invariants:} If an opponent's move can be perfectly mirrored/negated elsewhere, that component has zero impact on the game (e.g., ignore even steps in Staircase Nim).
    \end{itemize}
\end{enumerate}

\subsection{Sprague-Grundy Theorem}
\par Any independent sub-game can be replaced by an equivalent Nim pile of size $G$. The total Grundy value of a combined game state is the XOR sum ($\oplus$) of its sub-games:
$$G(\text{Game}) = G(\text{State}_1) \oplus G(\text{State}_2) \oplus \dots \oplus G(\text{State}_n)$$
\begin{itemize}
    \item $G(\text{Game}) \neq 0 \implies$ \textbf{Winning Position} (First player can force a win).
    \item $G(\text{Game}) = 0 \implies$ \textbf{Losing Position} (Second player wins under perfect play).
    \item $\text{MEX}(S)$ is the Minimum Excluded non-negative integer from the set of reachable states $S$.
\end{itemize}

\subsection{Nim Properties \& Variants}

\subsubsection{1. Standard Nim}
\par Game state is winning iff the XOR sum of all pile sizes $x_i$ is non-zero: $S = \bigoplus x_i \neq 0$.
\par \textbf{To make a winning move:} Find a pile $i$ where $(x_i \oplus S) < x_i$, then reduce that pile to exactly $x_i \oplus S$.

\subsubsection{2. Mis\`ere Nim (Last Take Loses)}
\par Play using standard Nim strategies until a move can force \textbf{all} remaining piles to have a size of exactly 1.
\begin{itemize}
    \item If all piles are size 1, the first player wins iff the number of piles is \textbf{even}.
    \item If at least one pile has size $> 1$, the first player wins iff standard $\bigoplus x_i \neq 0$.
\end{itemize}

\subsubsection{3. Subtraction Game}
\par Players can remove at most $k$ objects from a single pile.
$$G(x) = x \pmod{k+1}$$

int get_mex(const vector<int>& transitions) {
    vector<bool> vis(transitions.size() + 1, false);
    for (int x : transitions) if (x >= 0 && x <= (int)transitions.size()) vis[x] = true;
    for (int i = 0; ; ++i) if (!vis[i]) return i;
}
